## 🔧 Quick Code Examples

### iOS SDK Example

```swift
import PlaudDeviceSDK

class ViewController: UIViewController {
    var deviceAgent: PlaudDeviceAgent!
    
    override func viewDidLoad() {
        super.viewDidLoad()
        
        // Initialize SDK
        deviceAgent = PlaudDeviceAgent.shared
        deviceAgent.initSDK(
            hostName: "YourApp", 
            appKey: "YOUR_APP_KEY",
            bindToken: "YOUR_BIND_TOKEN",
            extra: [:]
        )
        
        // Set delegate
        deviceAgent.delegate = self
    }
    
    func startScanning() {
        deviceAgent.startScanBle()
    }
    
    func startRecording() {
        deviceAgent.startRecord()
    }
}

extension ViewController: PlaudDeviceAgentProtocol {
    func bleDeviceFound(_ device: BleDevice) {
        // Handle device discovery
    }
    
    func bleRecordStart(_ success: Bool) {
        // Handle recording start
    }
}
```

### Android SDK Example

```kotlin
import com.plaud.sdk.PlaudSDK

class MainActivity : AppCompatActivity() {
    private lateinit var bleAgent: BleAgent
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        // Initialize SDK
        val bleAgentListener = object : BleAgentListener {
            override fun onDeviceFound(device: BleDevice) {
                // Handle device discovery
            }
            
            override fun onRecordStart(success: Boolean) {
                // Handle recording start
            }
        }
        
        Sdk.initSdk(
            context = this,
            appKey = "YOUR_APP_KEY",
            bleAgentListener = bleAgentListener,
            hostName = "YourApp",
            extra = null
        )
        
        bleAgent = Sdk.bleAgent
    }
    
    fun startScanning() {
        bleAgent.scanBle(true)
    }
    
    fun startRecording() {
        bleAgent.startRecord("meeting", onRequest, onResponse)
    }
}
```
